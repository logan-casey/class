\begin{frame}{What does this paper do?}
\begin{wideitemize}
\item Auclert, Rognlie, Straub-style HA model in open-economy setting
\item Contractionary depreciations: two HA channels: real-income channel (import prices $\uparrow$ Y $\downarrow$) \textit{can} dominate Keynesian multiplier channel (foreign demand $\uparrow$ Y $\uparrow$)
\begin{itemize}
  \item These can together dominate RA-CM expenditure-switching channel (domestic home-goods demand $\uparrow$)
\end{itemize}
\item Quantitative model + calibration
\end{wideitemize}
\end{frame}


\begin{frame}{Base model}
\begin{wideitemize}
\item Domestic households face idiosyncratic income shocks and borrowing constraint
\item Households may hold domestic bond, foreign bond, domestic equity
\item Wages are sticky
\item IRFs to MIT shocks (perfect foresight, no aggregate uncertainty, etc.)
\end{wideitemize}
\end{frame}


\begin{frame}{Base model: simplifications}
\begin{wideitemize}
\item No illiquid asset/adjustment friction
\begin{itemize}
  \item UIP holds, portfolio choice indeterminate
\end{itemize}
\item No need to model consumption bundles
\item Rest-of-world is RA-CM-flexible price/wage (with patience/MP shock)
\item<2-> No capital; no taxes, debt, or spending
\item<2-> Producer-currency pricing (relaxed in appendix)
\end{wideitemize}
\end{frame}

\begin{frame}{Solution method: SSJ}
\begin{wideitemize}
\item Solve for steady state (all residuals = 0)
\item Compute sequence Jacobians (in background)
\begin{itemize}
  \item Must account for policy change, distribution change in HH block (as in PS2)
\end{itemize}
\item Solve for unknowns
\item Here:
\begin{itemize}
  \item Unknowns: $\{Y_t, W_t, Q_t, E_t, i_t, r_t\}$
  \item Targets: goods market clearing, NKPC, UIP, exchange rate, Fisher, MP equations
\end{itemize}
\end{wideitemize}
\end{frame}

\begin{frame}{Quantitative extension}
\begin{wideitemize}
\item Delayed substitution -- consumers change consumption basket with Calvo probability
\begin{itemize}
  \item Low trade elasticity in short run, high in long run
\end{itemize}
\item Heterogenous labor shock incidence (like \enquote{Inequality and Aggregate Demand})
\begin{itemize}
  \item Poor more affected
\end{itemize}
\item Non-homothetic consumption (Stone-Geary)
\begin{itemize}
  \item Poor consume more imports
\end{itemize}
\item Sticky prices
\end{wideitemize}
\end{frame}
